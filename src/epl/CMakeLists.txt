include("openroad")

set(THREADS_PREFER_PTHREAD_FLAG ON)

find_package(Eigen3 REQUIRED)
find_package(OpenMP REQUIRED)

project(epl)

swig_lib(NAME      epl
         NAMESPACE epl
         I_FILE    src/EPlace.i
         SWIG_INCLUDES ${ODB_HOME}/src/swig/common
                       ${ODB_HOME}/src/swig/tcl
                       ${ODB_HOME}/include
         SCRIPTS   src/EPlace.tcl
)

target_sources(epl
  PRIVATE
    src/MakeEPlace.cpp
    src/EPlace.cpp
    src/EDensity.cpp
    src/WAwirelength.cpp
)

messages(TARGET epl)

target_include_directories(epl
  PUBLIC
    include
  PRIVATE
    src
)

target_link_libraries(epl
    OpenSTA
    odb
    gpl
    Eigen3::Eigen
    Threads::Threads
    OpenMP::OpenMP_CXX
)
